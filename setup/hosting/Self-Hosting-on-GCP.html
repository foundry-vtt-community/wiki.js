:: Re-create the VM from snapshot
CALL gcloud compute instances create foundry-vm --project=foundry-390013 --zone=europe-west1-b --machine-type=e2-medium --network-interface=address=34.22.206.119,network-tier=PREMIUM,stack-type=IPV4_ONLY,subnet=foundrynet --maintenance-policy=MIGRATE --provisioning-model=STANDARD --service-account=768135181176-compute@developer.gserviceaccount.com --scopes=https://www.googleapis.com/auth/cloud-platform --tags=foundry-vm,http-server,https-server --create-disk=auto-delete=yes,boot=yes,device-name=foundry-vm,mode=rw,size=64,source-snapshot=projects/foundry-390013/global/snapshots/foundry-snapshot,type=projects/foundry-390013/zones/europe-west1-b/diskTypes/pd-balanced --metadata-from-file=startup-script=startup_script.txt --no-shielded-secure-boot --shielded-vtpm --shielded-integrity-monitoring --labels=goog-ec-src=vm_add-gcloud --reservation-affinity=any 

::Delete old archive snapshot
CALL gcloud compute snapshots delete foundry-archive-snapshot

::Create archive snapshot
CALL gcloud compute snapshots create foundry-archive-snapshot --project=foundry-390013 --source-disk=foundry-vm --source-disk-zone=europe-west1-b --storage-location=eu
